service cloud.firestore {
  match /databases/{database}/documents {

    // =========================
    // 🔐 FUNÇÕES BASE
    // =========================
    function isAuthenticated() {
      return request.auth != null;
    }
    function getUser() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }
    function userExists() {
      return exists(/databases/$(database)/documents/users/$(request.auth.uid));
    }
    function userTenant() {
      // FIX: campo é tenantSlug, não tenantId
      return getUser().tenantSlug;
    }
    function userRole() {
      return getUser().role;
    }
    function isMaster() {
      // FIX: role é 'master' (minúsculo), não 'SUPER_ADMIN'
      return userRole() == "master";
    }
    function isAdmin() {
      return userRole() == "admin" || isMaster();
    }
    function sameTenant(tenantId) {
      return userTenant() == tenantId || isMaster();
    }

    // =========================
    // 👤 USERS
    // =========================
    match /users/{userId} {
      allow read: if isAuthenticated()
        && userExists()
        && (request.auth.uid == userId || isMaster());
      allow create: if isAuthenticated();
      allow update: if isAuthenticated()
        && (request.auth.uid == userId || isMaster())
        && (isMaster() || (
          request.resource.data.tenantSlug == resource.data.tenantSlug
          && request.resource.data.role == resource.data.role
        ));
      allow delete: if isMaster();
    }

    // =========================
    // 🏢 TENANTS (CORE)
    // =========================
    match /tenants/{tenantId} {
      allow read: if isAuthenticated()
        && userExists()
        && (sameTenant(tenantId) || isMaster());
      allow create: if isMaster();
      allow update: if isMaster() || (isAuthenticated() && userExists() && sameTenant(tenantId));
      allow delete: if false;
    }

        && userExists()
        && (sameTenant(tenantId) || isMaster());
      allow write: if isAuthenticated()
        && userExists()
        && (sameTenant(tenantId) || isMaster());
    }
