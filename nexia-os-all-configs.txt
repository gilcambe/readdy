// ===== Firestore Security Rules =====
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Verifica se é o master
    function isMaster() {
      return exists(/databases/$(database)/documents/master_users/$(request.auth.uid));
    }

    // Obtém o tenantId do usuário logado
    function getTenantId() {
      return get(/databases/$(database)/documents/user_index/$(request.auth.uid)).data.tenantId;
    }

    // Regras por tenant — isolamento absoluto
    match /tenants/{tenantId} {
      allow read: if request.auth != null && (isMaster() || getTenantId() == tenantId);
      allow write: if request.auth != null && (isMaster() || getTenantId() == tenantId);

      // Conteúdo público (landing pages)
      match /content/{section} {
        allow read: if true;
        allow write: if request.auth != null && (isMaster() || getTenantId() == tenantId);
      }

      // Cadastros — qualquer um pode criar, só admin lê/edita
      match /cadastros/{doc} {
        allow create: if true;
        allow read, update, delete: if request.auth != null &&
          (isMaster() || getTenantId() == tenantId);
      }

      // Reservas (Splash)
      match /reservas/{doc} {
        allow create: if true;
        allow read, update, delete: if request.auth != null &&
          (isMaster() || getTenantId() == tenantId);
      }

      // Leilões (Bezsan)
      match /leiloes/{doc} {
        allow read: if true;
        allow write: if request.auth != null && (isMaster() || getTenantId() == tenantId);
      }
    }

    // Audit Log — IMUTÁVEL (só criação, nunca edição/deleção)
    match /_audit_logs/{doc} {
      allow create: if request.auth != null;
      allow read: if request.auth != null && isMaster();
      allow update, delete: if false;
    }
  }
}
