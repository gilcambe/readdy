 * POST /api/observe { action: "alert", ... }   → envia alerta

const admin = require('firebase-admin');
if (!admin.apps.length) {
  admin.initializeApp({ credential: admin.credential.cert(JSON.parse(process.env.FIREBASE_SERVICE_ACCOUNT)) });
}
const db  = admin.firestore();
const { guard, HEADERS } = require('./middleware');


// ── Cache em memória (shared entre warm invocations) ─────────
const _cache = new Map();
const CACHE_TTL = 30_000; // 30s


function cacheGet(key) {
  const entry = _cache.get(key);
  if (!entry) return null;
  if (Date.now() - entry.ts > CACHE_TTL) { _cache.delete(key); return null; }
  return entry.value;
}
function cacheSet(key, value) { _cache.set(key, { value, ts: Date.now() }); }


// ── Health check ──────────────────────────────────────────────
async function healthCheck() {
  const checks = {};
  const start = Date.now();


  // Firestore
  try {
    await db.collection('health').doc('ping').set({ ts: admin.firestore.FieldValue.serverTimestamp() });
    checks.firestore = { ok: true, ms: Date.now() - start };
  } catch (e) {
    checks.firestore = { ok: false, error: e.message };
  }


  // Groq API
  const groqStart = Date.now();
  try {
    const res = await fetch('https://api.groq.com/openai/v1/models', {
      headers: { 'Authorization': `Bearer ${process.env.GROQ_API_KEY}` }
    });
    checks.groq = { ok: res.ok, ms: Date.now() - groqStart, status: res.status };
  } catch (e) {
    checks.groq = { ok: false, error: e.message };
  }


  // DeepSeek
  if (process.env.DEEPSEEK_API_KEY) {
    const dsStart = Date.now();
    try {
      const res = await fetch('https://api.deepseek.com/models', {
        headers: { 'Authorization': `Bearer ${process.env.DEEPSEEK_API_KEY}` }
      });
      checks.deepseek = { ok: res.ok, ms: Date.now() - dsStart };
    } catch (e) {
      checks.deepseek = { ok: false, error: e.message };
    }
  }


  const allOk = Object.values(checks).every(c => c.ok);
  return {
    status: allOk ? 'healthy' : 'degraded',
    version: 'v10.0',
    timestamp: new Date().toISOString(),
    totalMs: Date.now() - start,
    checks
  };
}


// ── Métricas do tenant ────────────────────────────────────────
async function getMetrics(tenantId, hours = 24) {
  const cached = cacheGet(`metrics:${tenantId}:${hours}`);
  if (cached) return { ...cached, cached: true };


  const since = admin.firestore.Timestamp.fromDate(new Date(Date.now() - hours * 3600_000));


  const [execSnap, actionSnap, errSnap] = await Promise.all([
    db.collection('tenants').doc(tenantId).collection('cortex_logs')
      .where('type', '==', 'cortex_execution')
      .where('ts', '>=', since).get(),
    db.collection('tenants').doc(tenantId).collection('action_logs')
      .where('ts', '>=', since).get(),
    db.collection('tenants').doc(tenantId).collection('cortex_logs')
      .where('type', '==', 'cortex_error')
      .where('ts', '>=', since).get()
  ]);


  let totalMs = 0, count = 0;
  const intentBreakdown = {}, modelBreakdown = {}, activeUsers = new Set();
  const hourlyBuckets = {};


  for (const doc of execSnap.docs) {
    const d = doc.data();
    if (d.ms) { totalMs += d.ms; count++; }
    if (d.intent) intentBreakdown[d.intent] = (intentBreakdown[d.intent] || 0) + 1;
    if (d.modelUsed) modelBreakdown[d.modelUsed] = (modelBreakdown[d.modelUsed] || 0) + 1;
    if (d.userId) activeUsers.add(d.userId);


    // Hourly bucket
    const ts = d.ts?.toDate?.() || new Date();
    const hour = ts.toISOString().slice(0, 13);
    hourlyBuckets[hour] = (hourlyBuckets[hour] || 0) + 1;
  }


  const metrics = {
    tenantId, period: `${hours}h`,
    cortexCalls: execSnap.size,
    actionsCalled: actionSnap.size,
    errors: errSnap.size,
    errorRate: execSnap.size > 0 ? ((errSnap.size / execSnap.size) * 100).toFixed(1) + '%' : '0%',
    avgResponseMs: count > 0 ? Math.round(totalMs / count) : 0,
    p95ResponseMs: 0, // seria calculado com percentis reais
    activeUsers: activeUsers.size,
    intentBreakdown,
    modelBreakdown,
    hourlyBuckets: Object.entries(hourlyBuckets).map(([h, c]) => ({ hour: h, calls: c }))
  };


  cacheSet(`metrics:${tenantId}:${hours}`, metrics);
  return metrics;
}


// ── Alerta ────────────────────────────────────────────────────
async function sendAlert(tenantId, type, message, severity = 'info') {
  await db.collection('tenants').doc(tenantId).collection('alerts').add({
    type, message, severity, tenantId,
    ts: admin.firestore.FieldValue.serverTimestamp(),
    read: false
  });
  console.log(`[ALERT] [${severity.toUpperCase()}] ${tenantId}: ${message}`);
}


// ── HANDLER ───────────────────────────────────────────────────
exports.handler = async (event) => {
  if (event.httpMethod === 'OPTIONS') return { statusCode: 200, headers: HEADERS, body: '' };


  // Health é público
  const params = event.queryStringParameters || {};
  if (params.action === 'health') {
    const result = await healthCheck();
    return { statusCode: result.status === 'healthy' ? 200 : 503, headers: HEADERS, body: JSON.stringify(result) };
  }


  const guardErr = await guard(event, 'default');
  if (guardErr) return guardErr;


  try {
    if (event.httpMethod === 'GET') {
      const { tenantId, hours = '24' } = params;
      if (params.action === 'metrics') {
        if (!tenantId) return { statusCode: 400, headers: HEADERS, body: JSON.stringify({ error: 'tenantId obrigatório' }) };
        const metrics = await getMetrics(tenantId, parseInt(hours));
        return { statusCode: 200, headers: HEADERS, body: JSON.stringify(metrics) };
      }
    }


    if (event.httpMethod === 'POST') {
      const body = JSON.parse(event.body || '{}');
      if (body.action === 'alert') {
        await sendAlert(body.tenantId, body.type, body.message, body.severity);
        return { statusCode: 200, headers: HEADERS, body: JSON.stringify({ ok: true }) };
      }
    }


    return { statusCode: 400, headers: HEADERS, body: JSON.stringify({ error: 'Ação não reconhecida' }) };
  } catch (err) {
    return { statusCode: 500, headers: HEADERS, body: JSON.stringify({ error: err.message }) };
  }
};

// ── Stripe helper ─────────────────────────────────────────────
async function stripePost(endpoint, params) {
  const body = new URLSearchParams(params).toString();
  const res = await fetch(`${STRIPE_URL}/${endpoint}`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.STRIPE_SECRET_KEY}`,
      'Content-Type': 'application/x-www-form-urlencoded'
    },
    body
  });
  if (!res.ok) {
    const err = await res.json();
    throw new Error(err.error?.message || 'Stripe error');
  }
  return res.json();
}


async function stripeGet(endpoint) {
  const res = await fetch(`${STRIPE_URL}/${endpoint}`, {
    headers: { 'Authorization': `Bearer ${process.env.STRIPE_SECRET_KEY}` }
  });

// ── Processa Webhook Stripe ───────────────────────────────────
async function processWebhook(rawBody, signature) {
  // Verificação básica de assinatura (produção usa stripe.webhooks.constructEvent)
  // Aqui assumimos que a verificação é feita via HMAC se STRIPE_WEBHOOK_SECRET está definido
  let event;
  try { event = JSON.parse(rawBody); } catch { throw new Error('Body inválido'); }


  const { type, data } = event;
  console.log(`[BILLING] Webhook: ${type}`);


  if (type === 'checkout.session.completed') {
    const session = data.object;
    const { tenantId, plan } = session.metadata || {};
    if (!tenantId || !plan) return { ok: true, skipped: true };


    await db.collection('tenants').doc(tenantId).update({
      plan,
      stripeCustomerId: session.customer,
      stripeSubscriptionId: session.subscription,
      planActivatedAt: now(),
      updatedAt: now()
    });
    console.log(`[BILLING] Tenant "${tenantId}" upgraded to ${plan}`);
  }


  if (type === 'customer.subscription.deleted') {
    const sub = data.object;
    const tenantId = sub.metadata?.tenantId;
    if (tenantId) {
      await db.collection('tenants').doc(tenantId).update({
        plan: 'free', stripeSubscriptionId: null, updatedAt: now()
      });
      console.log(`[BILLING] Tenant "${tenantId}" downgraded to free`);
    }
  }


  if (type === 'invoice.payment_failed') {
    const invoice = data.object;
    const customerId = invoice.customer;
    // Busca tenant por customerId
    const snap = await db.collection('tenants').where('stripeCustomerId', '==', customerId).limit(1).get();
    if (!snap.empty) {
      const tenantId = snap.docs[0].id;
      await db.collection('tenants').doc(tenantId).update({
        billingStatus: 'payment_failed', updatedAt: now()
      });
    }
  }


  return { ok: true };
}

// ── Onboarding: cria tenant + define user como admin ─────────
async function registerTenant(uid, email, tenantName, tenantSlug) {
  // Valida slug
  const slug = tenantSlug.toLowerCase().replace(/[^a-z0-9-]/g, '-').slice(0, 30);


  // Verifica se slug já existe
  const existing = await db.collection('tenants').doc(slug).get();

  const batch = db.batch();


  // Cria tenant
  const tenantRef = db.collection('tenants').doc(slug);
  batch.set(tenantRef, {
    id: slug, name: tenantName, slug,
    plan: 'free', ownerId: uid,
    createdAt: now(), updatedAt: now(),
    settings: { language: 'pt-BR', timezone: 'America/Sao_Paulo' },
    ragDocsCount: 0,
    membersCount: 1
  });


  // Cria member record
  const memberRef = tenantRef.collection('members').doc(uid);
  batch.set(memberRef, { userId: uid, email, role: 'admin', joinedAt: now() });


  // Atualiza perfil do user
  const userRef = db.collection('users').doc(uid);
  batch.update(userRef, {
    tenantSlug: slug, role: 'admin',
    onboarded: true, updatedAt: now()
  });


  await batch.commit();


  return { tenantId: slug, role: 'admin' };
}


// ── Convida membro ─────────────────────────────────────────────
async function inviteMember(tenantId, inviterUid, inviteEmail, role = 'member') {
  // Verifica se inviter é admin
  const memberDoc = await db.collection('tenants').doc(tenantId).collection('members').doc(inviterUid).get();
  if (!memberDoc.exists) throw new Error('Inviter não é membro deste tenant');
  const inviterRole = memberDoc.data().role;
  if (!['master', 'admin'].includes(inviterRole)) throw new Error('Apenas admins podem convidar membros');


  // Cria convite
  const inviteRef = await db.collection('tenants').doc(tenantId).collection('invites').add({
    email: inviteEmail, role, invitedBy: inviterUid,
    status: 'pending', tenantId,
    createdAt: now(),
    expiresAt: admin.firestore.Timestamp.fromDate(new Date(Date.now() + 7 * 24 * 3600_000))
  });


  return { inviteId: inviteRef.id, email: inviteEmail, role };
}


// ── Aceita convite ────────────────────────────────────────────
async function acceptInvite(uid, email, inviteId) {
  const inviteRef = db.collection('tenants').doc('*'); // será resolvido por query


  // Busca convite por id em todos os tenants (via collectionGroup)
  const snap = await db.collectionGroup('invites').where(admin.firestore.FieldPath.documentId(), '==', inviteId).get();
  if (snap.empty) throw new Error('Convite não encontrado');


  const inviteDoc = snap.docs[0];
  const invite    = inviteDoc.data();
  if (invite.status !== 'pending') throw new Error('Convite já foi usado ou expirou');
  if (invite.email !== email) throw new Error('Este convite não pertence a este email');


  const now_ = now();
  const batch = db.batch();


  // Cria member
  const memberRef = db.collection('tenants').doc(invite.tenantId).collection('members').doc(uid);
  batch.set(memberRef, { userId: uid, email, role: invite.role, joinedAt: now_ });


  // Atualiza user
  const userRef = db.collection('users').doc(uid);
  batch.update(userRef, { tenantSlug: invite.tenantId, role: invite.role, onboarded: true, updatedAt: now_ });


  // Marca convite como aceito
  batch.update(inviteDoc.ref, { status: 'accepted', acceptedAt: now_, acceptedBy: uid });


  // Incrementa membersCount
  batch.update(db.collection('tenants').doc(invite.tenantId), {
    membersCount: admin.firestore.FieldValue.increment(1), updatedAt: now_
  });


  await batch.commit();
  return { tenantId: invite.tenantId, role: invite.role };
}


// ── HANDLER ───────────────────────────────────────────────────
exports.handler = async (event) => {
  if (event.httpMethod === 'OPTIONS') return { statusCode: 200, headers: HEADERS, body: '' };
  if (event.httpMethod !== 'POST') return { statusCode: 405, headers: HEADERS, body: 'Method Not Allowed' };


  try {
    const body = JSON.parse(event.body || '{}');
    const { action, token, userId, email, displayName } = body;


    // Verifica token em todas as ações exceto verify público
    let decoded = null;
    if (token) {
      try { decoded = await verifyToken(token); } catch (e) {
        return { statusCode: 401, headers: HEADERS, body: JSON.stringify({ error: 'Token inválido ou expirado' }) };
      }
    }


    const uid = decoded?.uid || userId;


    if (action === 'verify') {
      if (!decoded) return { statusCode: 401, headers: HEADERS, body: JSON.stringify({ error: 'Token necessário' }) };
      const profile = await getOrCreateProfile(decoded.uid, decoded.email, decoded.name);
      return { statusCode: 200, headers: HEADERS, body: JSON.stringify({ uid: decoded.uid, ...profile }) };
    }


    if (action === 'register') {
      const { tenantName, tenantSlug } = body;
      if (!uid || !tenantName || !tenantSlug) throw new Error('uid, tenantName e tenantSlug são obrigatórios');
      const result = await registerTenant(uid, email || decoded?.email, tenantName, tenantSlug);
      return { statusCode: 200, headers: HEADERS, body: JSON.stringify({ ok: true, ...result }) };
    }


    if (action === 'profile') {
      if (!uid) return { statusCode: 400, headers: HEADERS, body: JSON.stringify({ error: 'userId obrigatório' }) };
      const profile = await getOrCreateProfile(uid, email, displayName);
      return { statusCode: 200, headers: HEADERS, body: JSON.stringify(profile) };
    }


    if (action === 'invite') {
      const { tenantId, inviteEmail, role } = body;
      const result = await inviteMember(tenantId, uid, inviteEmail, role);
      return { statusCode: 200, headers: HEADERS, body: JSON.stringify({ ok: true, ...result }) };
    }


    if (action === 'accept') {
      const { inviteId } = body;
      const result = await acceptInvite(uid, decoded?.email || email, inviteId);
      return { statusCode: 200, headers: HEADERS, body: JSON.stringify({ ok: true, ...result }) };
    }


    return { statusCode: 400, headers: HEADERS, body: JSON.stringify({ error: 'Ação não reconhecida' }) };


  } catch (err) {
    console.error('[AUTH] ❌', err.message);
    return { statusCode: 500, headers: HEADERS, body: JSON.stringify({ error: err.message }) };
  }
};

// ── Busca stats de uso ─────────────────────────────────────────
async function getUsageStats(tenantId, days = 7) {
  const since = new Date();
  since.setDate(since.getDate() - days);


  const snap = await db.collection('tenants').doc(tenantId)
    .collection('usage')
    .where('date', '>=', since.toISOString().split('T')[0])
    .orderBy('date', 'desc')
    .limit(days + 1)
    .get();


  const daily = snap.docs.map(d => ({ date: d.id, ...d.data() }));
  const totalCalls = daily.reduce((acc, d) => acc + (d.cortexCalls || 0), 0);
  const today = new Date().toISOString().split('T')[0];
  const todayData = daily.find(d => d.date === today);
  const todayCalls = todayData?.cortexCalls || 0;


  // Pega plano do tenant
  const tenantDoc = await db.collection('tenants').doc(tenantId).get();
  const plan = tenantDoc.exists ? (tenantDoc.data().plan || 'free') : 'free';
  const limits = PLAN_LIMITS[plan] || PLAN_LIMITS.free;


  return {
    tenantId, plan, limits,
    today: { calls: todayCalls, limit: limits.maxCortexDay, pct: limits.maxCortexDay > 0 ? Math.round(todayCalls / limits.maxCortexDay * 100) : 0 },
    period: { days, totalCalls, daily },
    features: {
      swarmEnabled: limits.swarmEnabled,
      autodevEnabled: limits.autodevEnabled,
      ragDocsLeft: limits.maxRAGDocs > 0 ? Math.max(0, limits.maxRAGDocs - (tenantDoc.data()?.ragDocsCount || 0)) : -1
    }
  };
}


// ── Verifica se pode chamar o Cortex ──────────────────────────
async function checkLimit(tenantId, userId, feature = 'cortex') {
  const today = new Date().toISOString().split('T')[0];
  const [usageDoc, tenantDoc] = await Promise.all([
    db.collection('tenants').doc(tenantId).collection('usage').doc(today).get(),
    db.collection('tenants').doc(tenantId).get()
  ]);


  const plan   = tenantDoc.exists ? (tenantDoc.data().plan || 'free') : 'free';
  const limits = PLAN_LIMITS[plan] || PLAN_LIMITS.free;


  if (feature === 'swarm' && !limits.swarmEnabled) {
    return { ok: false, reason: `Swarm não disponível no plano ${plan}. Faça upgrade para Starter.`, upgradeRequired: true };
  }
  if (feature === 'autodev' && !limits.autodevEnabled) {
    return { ok: false, reason: `AutoDev não disponível no plano ${plan}. Faça upgrade para Pro.`, upgradeRequired: true };
  }


  const todayCalls = usageDoc.exists ? (usageDoc.data()?.cortexCalls || 0) : 0;
  const maxDay = limits.maxCortexDay;


  if (maxDay > 0 && todayCalls >= maxDay) {
    return {
      ok: false,
      reason: `Limite diário atingido: ${todayCalls}/${maxDay} mensagens (plano ${plan}).`,
      upgradeRequired: true,
      current: todayCalls, max: maxDay, plan
    };
  }


  return { ok: true, remaining: maxDay > 0 ? maxDay - todayCalls : -1, plan };
}

  return callAI('Return ONLY corrected valid JSON. No text outside JSON.', [{ role: 'user', content: broken }],
    { model: 'llama-3.1-8b-instant', temperature: 0, max_tokens: 600 });
}
async function parseOrchestrator(raw) {
  let d = safeParseJSON(raw);
  if (d) return { decision: d, layer: 1 };
  const ex = extractJSON(raw);
  if (ex) { d = safeParseJSON(ex); if (d) return { decision: d, layer: 2 }; }
  try {
    const rep = await repairJSON(raw);
    d = safeParseJSON(rep) || safeParseJSON(extractJSON(rep) || '');
    if (d) return { decision: d, layer: 3 };
  } catch {}
  return { decision: { type: 'chat', response: raw }, layer: 'fallback' };
}


// ── Swarm ─────────────────────────────────────────────────────
async function runSwarm(agentNames, messages, dynamicAgents = {}) {
  const allAgents = { ...STATIC_AGENTS, ...dynamicAgents };
  const results = await Promise.all(
    agentNames.filter(n => allAgents[n]).map(async name => {
      try {
        const reply = await callAI(allAgents[name].system, messages, { temperature: 0.5, max_tokens: 800 });
        return { name, reply, ok: true };
      } catch (e) { return { name, reply: `[Agente ${name} indisponível]`, ok: false }; }
    })
  );
  if (results.filter(r => r.ok).length > 1) {
    const summaryInput = results.map(r => `### ${r.name.toUpperCase()}\n${r.reply}`).join('\n\n---\n\n');
    try {
      const synthesis = await callAI(
        'Consolide as análises dos especialistas em uma resposta unificada, executiva e em português.',
        [{ role: 'user', content: summaryInput }], { temperature: 0.4, max_tokens: 1000 }
      );
      return { outputs: results, synthesis };
    } catch {}
  }
  return { outputs: results, synthesis: results[0]?.reply || '' };
}


// ── Agentes dinâmicos por tenant ──────────────────────────────
async function loadDynamicAgents(tenantId) {
  try {
    const [tenantSnap, globalSnap] = await Promise.all([
      db.collection('agents').where('tenantId', '==', tenantId).where('active', '!=', false).get(),
      db.collection('agents').where('global', '==', true).get()
    ]);
    const agents = {};
    [...tenantSnap.docs, ...globalSnap.docs].forEach(d => {
      const a = d.data();
      if (a.systemPrompt) agents[d.id] = { system: a.systemPrompt };
    });
    return agents;
  } catch { return {}; }
}


// ── Tracking de uso + verificação de limite ───────────────────
async function trackUsage(tenantId, userId) {
  const today = new Date().toISOString().split('T')[0];
  const usageRef = db.collection('tenants').doc(tenantId).collection('usage').doc(today);
  try {
    const [, tenantDoc] = await Promise.all([
      db.runTransaction(async tx => {
        const doc = await tx.get(usageRef);
        if (!doc.exists) {
          tx.set(usageRef, { date: today, cortexCalls: 1, tenantId, userBreakdown: { [userId]: 1 } });
        } else {
          tx.update(usageRef, {
            cortexCalls: admin.firestore.FieldValue.increment(1),
            [`userBreakdown.${userId}`]: admin.firestore.FieldValue.increment(1)
          });
        }
      }),
      db.collection('tenants').doc(tenantId).get()
    ]);
    const plan = tenantDoc.exists ? (tenantDoc.data().plan || 'free') : 'free';
    const LIMITS = { free: 20, starter: 200, pro: 1000, enterprise: -1 };
    const limit = LIMITS[plan] ?? 20;
    if (limit > 0) {
      const usageDoc = await usageRef.get();
      const calls = usageDoc.data()?.cortexCalls || 0;
      if (calls > limit) throw new Error(`Limite diário do plano "${plan}" atingido (${limit} msgs/dia). Faça upgrade para continuar.`);
    }
  } catch (e) {
    if (e.message?.includes('Limite')) throw e;
    console.warn('[CORTEX] Usage tracking error:', e.message);
  }
}


// ── RAG context injection ─────────────────────────────────────
async function buildRAGSystemAddition(tenantId, message) {
  try {
    const ragModule = require('./rag-engine');
    if (typeof ragModule.buildRAGContext !== 'function') return '';
    const ctx = await ragModule.buildRAGContext(tenantId, message);
    return ctx ? `\n\n[CONTEXTO DE DOCUMENTOS]\n${ctx}\n[FIM DO CONTEXTO]` : '';
  } catch { return ''; }
}


// ── Log estruturado ───────────────────────────────────────────
async function log(tenantId, userId, type, data) {
  try {
    await db.collection('tenants').doc(tenantId).collection('cortex_logs').add({
      type, userId, ...data, ts: admin.firestore.FieldValue.serverTimestamp()
    });
  } catch {}
}


// ── HANDLER ───────────────────────────────────────────────────
exports.handler = async (event) => {
  if (event.httpMethod === 'OPTIONS') return { statusCode: 200, headers: HEADERS, body: '' };
  if (event.httpMethod !== 'POST') return { statusCode: 405, headers: HEADERS, body: 'Method Not Allowed' };
  const guardErr = await guard(event, 'cortex-chat');
  if (guardErr) return guardErr;


  const start = Date.now();
  try {
    const { userId, message: rawMessage, tenantId = 'nexia', ragEnabled = false, model: modelOverride } = JSON.parse(event.body || '{}');
    if (!userId || !rawMessage) throw new Error('userId e message são obrigatórios');


    const message = sanitizePrompt(rawMessage);
    await trackUsage(tenantId, userId);


    const mem     = await memModule.load(userId, tenantId);
    const context = memModule.buildContext(mem.history, mem.summaries, 20);
    const learningCtx = await learnModule.buildLearningContext(tenantId, message).catch(() => null);
    const ragAddition = ragEnabled ? await buildRAGSystemAddition(tenantId, message) : '';


    const decisionContext = [
      ...(learningCtx ? [learningCtx] : []),
      ...context,
      { role: 'user', content: message }
    ];


    const decisionRaw = await callAI(STATIC_AGENTS.orchestrator.system + ragAddition, decisionContext.slice(-15));
    const { decision, layer } = await parseOrchestrator(decisionRaw);


    console.log(`[CORTEX v10] layer:${layer} type:${decision.type} tenant:${tenantId} rag:${ragEnabled}`);


    let finalResponse = decision.response || '';
    const executedActions = [];
    let modelUsed = 'groq-orchestrator';


    // Ações
    if (Array.isArray(decision.actions) && decision.actions.length > 0) {
      const userRole = event._role || 'user';
      for (const act of decision.actions) {
        try {
          if (!act.type || !act.data) continue;
          validateAIAction(act.type, act.data);
          if (!checkPermission(userRole, act.type)) {
            executedActions.push({ ok: false, action: act.type, error: `Role "${userRole}" sem permissão para "${act.type}"` });
            continue;
          }
          const result = await actionModule.dispatch(act.type, act.data, tenantId, userId);
          executedActions.push(result);
        } catch (e) {
          executedActions.push({ ok: false, error: e.message, action: act.type });
        }
      }
      if (executedActions.length) {
        const lines = executedActions.map(r => r.ok ? `✅ \`${r.action}\` executado (ID: \`${r.id}\`)` : `⚠️ \`${r.action}\` falhou: ${r.error}`).join('\n');
        finalResponse += finalResponse ? `\n\n${lines}` : lines;
      }
    }


    // Swarm
    if (decision.type === 'swarm' && Array.isArray(decision.agents) && decision.agents.length) {
      const dynamicAgents = await loadDynamicAgents(tenantId);
      const swarm = await runSwarm(decision.agents, decisionContext.slice(-10), dynamicAgents);
      finalResponse += finalResponse ? `\n\n${swarm.synthesis}` : swarm.synthesis;
      modelUsed = 'swarm';
    }


    // Chat/Code via multi-model
    if (!finalResponse.trim()) {
      const intent = decision.model_hint || (decision.type === 'code' ? 'code' : 'chat');
      try {
        const result = await callWithFallback({
          system: `Você é o CORTEX NEXIA — IA executiva. Responda em português.${ragAddition}`,
          messages: decisionContext.slice(-15),
          intent
        });
        finalResponse = typeof result === 'object' ? result.response : result;
        modelUsed = typeof result === 'object' ? (result.model || intent) : intent;
      } catch {
        finalResponse = await callAI(
          `Você é o CORTEX NEXIA. Responda em português.${ragAddition}`,
          decisionContext.slice(-15), { temperature: 0.4 }
        );
      }
    }


    // Salva memória
    const newMessages = [{ role: 'user', content: message }, { role: 'assistant', content: finalResponse }];
    const updatedHistory = [...mem.history, ...newMessages];
    const entities = memModule.extractEntities(newMessages, mem.entities);
    await memModule.save(userId, updatedHistory, mem.summaries, tenantId, entities);


    learnModule.saveExample && learnModule.saveExample(tenantId, userId, message, finalResponse, decision.type).catch(() => {});


    const ms = Date.now() - start;
    await log(tenantId, userId, 'cortex_execution', {
      intent: decision.type, layer, ms, actionsCount: executedActions.length,
      agentsUsed: decision.agents || [], modelUsed, ragEnabled
    });


    return {
      statusCode: 200,
      headers: HEADERS,
      body: JSON.stringify({
        reply: finalResponse,
        intent: decision.type,
        actions: executedActions,
        _meta: { layer, ms, modelUsed, ragEnabled, version: 'v10' }
      })
    };


  } catch (err) {
    console.error('[CORTEX v10] ❌ Erro crítico:', err.message);
    return {
      statusCode: err.message?.includes('Limite') ? 429 : err.message?.includes('não permitid') ? 400 : 500,
      headers: HEADERS,
      body: JSON.stringify({ error: err.message })
    };
  }
};
